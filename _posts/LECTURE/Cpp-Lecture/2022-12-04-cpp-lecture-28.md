  int Number::getDec()

      int i = 1;
      int num = 0;
      for (int m = n.size() - 1; m >= 0; m--)
      {
          num += (n[m] - '0') * i;
          i *= k;
      }
      return num;
  }

  10진수에서 k진수로 바꾸기

  ```cpp
  // 10진수에서 k진수로 바꾸기
  void Number::changeToK(int k)
  {
      int dec = getDec();
      this->k = k;
      int i = 1;
      this->n = "";
      for (int m = dec; m > 0; m /= k)
      {
          this->n = (char)('0' + (m % k)) + this->n;
      }
  }
  ```
